148 class TopicList {
149  // Contains a list of topics which require creation.
150  //
151  // Created from topic strings or from TOPIC_SUBSCRIBED or
152  // RESOLUTION_SUCCESS messages.
153  // This is passed to a createTopics() call or createTopicsAsync()
154  // call on a ProviderSession. It is updated and returned by the
155  // createTopics() call.
156 
157  blpapi_TopicList_t *d_handle_p;
158 
159  public:
160  enum Status {
161  NOT_CREATED,
162  CREATED,
163  FAILURE
164  };
165 
166  // CLASS METHODS
167  TopicList();
168  // Create an empty TopicList.
169 
170  TopicList(const ResolutionList& original);
171  // Create a topic list from ResolutionList.
172  // User has previously resolved topics in ResolutionList
173  // and wants to now createTopics.
174 
175  TopicList(const TopicList& original);
176  // Copy constructor.
177 
178  ~TopicList();
179  // Destroy this TopicList.
180 
181  // MANIPULATORS
182  int add(const char* topic,
183  const CorrelationId& correlationId=CorrelationId());
184  // Add the specified 'topic' to this list, optionally
185  // specifying a 'correlationId'. Returns 0 on success or
186  // negative number on failure. After a successful call to
187  // add() the status for this entry is NOT_CREATED.
188 
189  int add(Message const& message,
190  const CorrelationId& correlationId=CorrelationId());
191  // Add the topic contained in the specified
192  // 'topicSubscribedMessage' or 'resolutionSuccessMessage'
193  // to this list, optionally specifying a 'correlationId'.
194  // Returns 0 on success or a negative number on failure.
195  // After a successful call to add()
196  // the status for this entry is NOT_CREATED.
197 
198  // ACCESSORS
199  CorrelationId correlationIdAt(size_t index) const;
200  // Returns the CorrelationId of the specified 'index'th entry in
201  // this TopicList. If 'index' >= size() an exception is
202  // thrown.
203 
204  const char* topicString(const CorrelationId& correlationId) const;
205  // Returns a pointer to the topic of the entry identified by
206  // the specified 'correlationId'. If the 'correlationId' does
207  // not identify an entry in this TopicList then an
208  // exception is thrown.
209 
210  const char* topicStringAt(size_t index) const;
211  // Returns a pointer to the topic of the specified 'index'th
212  // entry. If 'index' >= size() an exception is thrown.
213 
214  int status(const CorrelationId& correlationId) const;
215  // Returns the status of the entry in this TopicList
216  // identified by the specified 'correlationId'. This may be
217  // NOT_CREATED, CREATED and FAILURE.
218  // If the 'correlationId' does not identify an entry in this
219  // TopicList then an exception is thrown.
220 
221  int statusAt(size_t index) const;
222  // Returns the status of the specified 'index'th entry in this
223  // TopicList. This may be NOT_CREATED, CREATED and FAILURE.
224  // If 'index' > size() an exception is thrown.
225 
226  Message const message(const CorrelationId& correlationId) const;
227  // Returns the value of the message received during
228  // creation of the topic identified by the specified
229  // 'correlationId'. If 'correlationId' does not identify an
230  // entry in this TopicList or if the status of the entry
231  // identify by 'correlationId' is not CREATED an exception is
232  // thrown.
233  //
234  // The message returned can be used when creating an instance
235  // of Topic.
236 
237  Message const messageAt(size_t index) const;
238  // Returns the value of the message received during creation
239  // of the specified 'index'th entry in this TopicList. If
240  // 'index' >= size() or if the status of the 'index'th entry
241  // is not CREATED an exception is thrown.
242  //
243  // The message returned can be used when creating an instance
244  // of Topic.
245 
246  size_t size() const;
247  // Returns the number of entries in this list.
248 
249  const blpapi_TopicList_t* impl() const;
