123 class Event {
124  // A single event resulting from a subscription or a request.
125  //
126  // Event objects are created by the API and passed to the
127  // application either through a registered EventHandler or
128  // EventQueue or returned from the Session::nextEvent()
129  // method. Event objects contain Message objects which can be
130  // accessed using a MessageIterator.
131  //
132  // The Event object is a handle to an event. The event is the
133  // basic unit of work provided to applications. Each Event object
134  // consists of an EventType attribute and zero or more Message
135  // objects. The underlying event data including the messages is
136  // reference counted - as long as at least one Event object still
137  // exists then the underlying data will not be freed.
138 
139  blpapi_Event_t *d_impl_p;
140 
141  public:
142  enum EventType {
143  // The possible types of event
144  ADMIN = BLPAPI_EVENTTYPE_ADMIN,
145  // Admin event
146  SESSION_STATUS = BLPAPI_EVENTTYPE_SESSION_STATUS,
147  // Status updates for a session.
148  SUBSCRIPTION_STATUS = BLPAPI_EVENTTYPE_SUBSCRIPTION_STATUS,
149  // Status updates for a subscription.
150  REQUEST_STATUS = BLPAPI_EVENTTYPE_REQUEST_STATUS,
151  // Status updates for a request.
152  RESPONSE = BLPAPI_EVENTTYPE_RESPONSE,
153  // The final (possibly only) response to a request.
154  PARTIAL_RESPONSE = BLPAPI_EVENTTYPE_PARTIAL_RESPONSE,
155  // A partial response to a request.
156  SUBSCRIPTION_DATA = BLPAPI_EVENTTYPE_SUBSCRIPTION_DATA,
157  // Data updates resulting from a subscription.
158  SERVICE_STATUS = BLPAPI_EVENTTYPE_SERVICE_STATUS,
159  // Status updates for a service.
160  TIMEOUT = BLPAPI_EVENTTYPE_TIMEOUT,
161  // An Event returned from nextEvent() if it timed out.
162  AUTHORIZATION_STATUS = BLPAPI_EVENTTYPE_AUTHORIZATION_STATUS,
163  // Status updates for user authorization.
164  RESOLUTION_STATUS = BLPAPI_EVENTTYPE_RESOLUTION_STATUS,
165  // Status updates for a resolution operation.
166  TOPIC_STATUS = BLPAPI_EVENTTYPE_TOPIC_STATUS,
167  // Status updates about topics for service providers.
168  TOKEN_STATUS = BLPAPI_EVENTTYPE_TOKEN_STATUS,
169  // Status updates for a generate token request.
170  REQUEST = BLPAPI_EVENTTYPE_REQUEST,
171  // Request event
172  UNKNOWN = -1
173  };

214 class EventQueue {
215  // A construct used to handle replies to request synchronously.
216  //
217  // An EventQueue can be supplied when using Session::sendRequest()
218  // and Session::sendAuthorizationRequest() methods.
219  //
220  // When a request is submitted an application can either handle
221  // the responses asynchronously as they arrive or use an
222  // EventQueue to handle all responses for a given request or
223  // requests synchronously. The EventQueue will only deliver
224  // responses to the request(s) it is associated with.
225 
226  blpapi_EventQueue_t *d_handle_p;
227 
228  public:
229  EventQueue();
230  // Construct an empty event queue.
231 
232  ~EventQueue();
233  // Destroy this event queue and cancel any pending request
234  // that are linked to this queue.
235 
236  // MANIPULATORS
237 
238  Event nextEvent(int timeout=0);
239  // Returns the next Event available from the EventQueue. If
240  // the specified 'timeout' is zero this will wait forever for
241  // the next event. If the specified 'timeout' is non zero then
242  // if no Event is available within the specified 'timeout' an
243  // Event with a type() of TIMEOUT will be returned.
244 
245  int tryNextEvent(Event *event);
246  // If the EventQueue is non-empty, load the next Event available
247  // into event and return 0 indicating success. If the EventQueue is
248  // empty, return a non-zero value with no effect on event or the
249  // the state of EventQueue. This method never blocks.

254  // this EventQueue. The EventQueue can subsequently be
