153  bool processSubscriptionStatus(const Event &event, Session *session)
154  {
155  char timeBuffer[64];
156  getTimeStamp(timeBuffer, sizeof(timeBuffer));
157 
158  SubscriptionList subscriptionList;
159  ConsoleOut(d_consoleLock_p)
160  << "\nProcessing SUBSCRIPTION_STATUS"
161  << std::endl;
162 
163  MessageIterator msgIter(event);
164  while (msgIter.next()) {
165  Message msg = msgIter.message();
166  CorrelationId cid = msg.correlationId();
167 
168  std::string topic = getTopic(cid);
169  {
170  ConsoleOut out(d_consoleLock_p);
171  out << timeBuffer << ": " << topic
172  << std::endl;
173  msg.print(out.stream(), 0, 4);
174  }
175 
176  if (msg.messageType() == SUBSCRIPTION_TERMINATED
177  && d_pendingUnsubscribe.erase(cid)) {
178  // If this message was due to a previous unsubscribe
179  const char *topicString = getSubscriptionTopicString(
180  d_context_p->d_subscriptions,
181  cid);
182  assert(topicString);
183  if (d_isSlow) {
184  ConsoleOut(d_consoleLock_p)
185  << "Deferring subscription for topic = " << topic
186  << " because session is slow." << std::endl;
187  d_pendingSubscriptions.add(topicString, cid);
188  }
189  else {
190  subscriptionList.add(topicString, cid);
191  }
192  }
193  }
194 
195  if (0 != subscriptionList.size() && !d_context_p->d_isStopped) {
196  session->subscribe(subscriptionList);
197  }
198 
199  return true;
200  }

222  bool processAdminEvent(const Event &event, Session *session)
223  {
224  char timeBuffer[64];
225  getTimeStamp(timeBuffer, sizeof(timeBuffer));
226 
227  ConsoleOut(d_consoleLock_p)
228  << "\nProcessing ADMIN" << std::endl;
229  std::vector<CorrelationId> cidsToCancel;
230  bool previouslySlow = d_isSlow;
231  MessageIterator msgIter(event);
232 
233  while (msgIter.next()) {
234  Message msg = msgIter.message();
235  // An admin event can have more than one messages.
236  if (msg.messageType() == DATA_LOSS) {
237  const CorrelationId& cid = msg.correlationId();
238  {
239  ConsoleOut out(d_consoleLock_p);
240  out << timeBuffer << ": " << getTopic(msg.correlationId())
241  << std::endl;
242  msg.print(out.stream(), 0, 4);
243  }
244 
245  if (msg.hasElement(SOURCE)) {
246  std::string sourceStr = msg.getElementAsString(SOURCE);
247  if (0 == sourceStr.compare("InProc")
248  && d_pendingUnsubscribe.find(cid) ==
249  d_pendingUnsubscribe.end()) {
250  // DataLoss was generated "InProc".
251  // This can only happen if applications are processing
252  // events slowly and hence are not able to keep-up with
253  // the incoming events.
254  cidsToCancel.push_back(cid);
255  d_pendingUnsubscribe.insert(cid);
256  }
257  }
258  }
259  else {
260  ConsoleOut(d_consoleLock_p)
261  << timeBuffer << ": "
262  << msg.messageType().string() << std::endl;
263  if (msg.messageType() == SLOW_CONSUMER_WARNING) {
264  d_isSlow = true;
265  }
266  else if (msg.messageType() == SLOW_CONSUMER_WARNING_CLEARED) {
267  d_isSlow = false;
268  }
269  }
270  }
271  if (!d_context_p->d_isStopped) {
272  if (0 != cidsToCancel.size()) {
273  session->cancel(cidsToCancel);
274  }
275  else if (previouslySlow
276  && !d_isSlow
277  && d_pendingSubscriptions.size() > 0) {
278  // Session was slow but is no longer slow. subscribe to any
279  // topics for which we have previously received
280  // SUBSCRIPTION_TERMINATED
281  ConsoleOut(d_consoleLock_p)
282  << "Subscribing to topics - "
283  << getTopicsString(d_pendingSubscriptions)
284  << std::endl;
285  session->subscribe(d_pendingSubscriptions);
286  d_pendingSubscriptions.clear();
287  }
288  }
289 
290  return true;
291  }

316  bool processEvent(const Event &event, Session *session)
317  {
318  try {
319  switch (event.eventType()) {
320  case Event::SUBSCRIPTION_DATA: {
321  return processSubscriptionDataEvent(event);
322  } break;
323  case Event::SUBSCRIPTION_STATUS: {
324  MutexGuard guard(&d_context_p->d_mutex);

327  case Event::ADMIN: {
328  MutexGuard guard(&d_context_p->d_mutex);
329  return processAdminEvent(event, session);
330  } break;
331  default: {
332  return processMiscEvents(event);

334  }
335  } catch (Exception &e) {

339  }
