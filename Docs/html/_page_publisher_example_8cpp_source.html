162 bool MyEventHandler::processEvent(const Event& event, ProviderSession* session)
163 {
164  if (event.eventType() == Event::SESSION_STATUS) {
165  printMessages(event);
166  MessageIterator iter(event);
167  while (iter.next()) {
168  Message msg = iter.message();
169  if (msg.messageType() == SESSION_TERMINATED) {
170  g_running = false;
171  }
172  }
173  }
174  else if (event.eventType() == Event::TOPIC_STATUS) {
175  TopicList topicList;
176  MessageIterator iter(event);
177  while (iter.next()) {
178  Message msg = iter.message();
179  std::cout << msg << std::endl;
180  if (msg.messageType() == TOPIC_SUBSCRIBED) {
181  std::string topicStr = msg.getElementAsString("topic");
182  MutexGuard guard(&g_mutex);
183  MyStreams::iterator it = g_streams.find(topicStr);
184  if (it == g_streams.end()) {
185  // TopicList knows how to add an entry based on a
186  // TOPIC_SUBSCRIBED message.
187  topicList.add(msg);
188  it = (g_streams.insert(MyStreams::value_type(
189  topicStr,
190  new MyStream(topicStr)))).first;
191  }
192  it->second->setSubscribedState(true);
193  if (it->second->isAvailable()) {
194  ++g_availableTopicCount;
195  }
196  }
197  else if (msg.messageType() == TOPIC_UNSUBSCRIBED) {
198  std::string topicStr = msg.getElementAsString("topic");
199  MutexGuard guard(&g_mutex);
200  MyStreams::iterator it = g_streams.find(topicStr);
201  if (it == g_streams.end()) {
202  // we should never be coming here. TOPIC_UNSUBSCRIBED can
203  // not come before a TOPIC_SUBSCRIBED or TOPIC_CREATED
204  continue;
205  }
206  if (it->second->isAvailable()) {
207  --g_availableTopicCount;
208  }
209  it->second->setSubscribedState(false);
210  }
211  else if (msg.messageType() == TOPIC_CREATED) {
212  std::string topicStr = msg.getElementAsString("topic");
213  MutexGuard guard(&g_mutex);
214  MyStreams::iterator it = g_streams.find(topicStr);
215  if (it == g_streams.end()) {
216  it = (g_streams.insert(MyStreams::value_type(
217  topicStr,
218  new MyStream(topicStr)))).first;
219  }
220  try {
221  Topic topic = session->getTopic(msg);
222  it->second->setTopic(topic);
223  } catch (blpapi::Exception &e) {
224  std::cerr
225  << "Exception in Session::getTopic(): "
226  << e.description()
227  << std::endl;
228  continue;
229  }
230  if (it->second->isAvailable()) {
231  ++g_availableTopicCount;
232  }
233  }
234  else if (msg.messageType() == TOPIC_RECAP) {
235  // Here we send a recap in response to a Recap request.
236  try {
237  std::string topicStr = msg.getElementAsString("topic");
238  MyStreams::iterator it = g_streams.find(topicStr);
239  MutexGuard guard(&g_mutex);
240  if (it == g_streams.end() || !it->second->isAvailable()) {
241  continue;
242  }
243  Topic topic = session->getTopic(msg);
244  Service service = topic.service();
245  CorrelationId recapCid = msg.correlationId();
246 
247  Event recapEvent = service.createPublishEvent();
248  EventFormatter eventFormatter(recapEvent);
249  eventFormatter.appendRecapMessage(topic, &recapCid);
250  eventFormatter.setElement("numRows", 25);
251  eventFormatter.setElement("numCols", 80);
252  eventFormatter.pushElement("rowUpdate");
253  for (int i = 1; i < 6; ++i) {
254  eventFormatter.appendElement();
255  eventFormatter.setElement("rowNum", i);
256  eventFormatter.pushElement("spanUpdate");
257  eventFormatter.appendElement();
258  eventFormatter.setElement("startCol", 1);
259  eventFormatter.setElement("length", 10);
260  eventFormatter.setElement("text", "RECAP");
261  eventFormatter.popElement();
262  eventFormatter.popElement();
263  eventFormatter.popElement();
264  }
265  eventFormatter.popElement();
266  guard.release()->unlock();
267  session->publish(recapEvent);
268  } catch (blpapi::Exception &e) {
269  std::cerr
270  << "Exception in Session::getTopic(): "
271  << e.description()
272  << std::endl;
273  continue;
274  }
275  }
276  }
277  if (topicList.size()) {
278  // createTopicsAsync will result in RESOLUTION_STATUS,
279  // TOPIC_CREATED events.
280  session->createTopicsAsync(topicList);
281  }
282  }
283  else if (event.eventType() == Event::RESOLUTION_STATUS) {
284  printMessages(event);
285  }
286  else if (event.eventType() == Event::REQUEST) {
287  Service service = session->getService(d_serviceName.c_str());
288  MessageIterator iter(event);
289  while (iter.next()) {
290  Message msg = iter.message();
291  msg.print(std::cout);
292  if (msg.messageType() == PERMISSION_REQUEST) {
293  // This example always sends a 'PERMISSIONED' response.
294  // See 'MktdataPublisherExample' on how to parse a Permission
295  // request and send an appropriate 'PermissionResponse'.
296  Event response = service.createResponseEvent(
297  msg.correlationId());
298  int permission = 0; // ALLOWED: 0, DENIED: 1
299  EventFormatter ef(response);
300  ef.appendResponse("PermissionResponse");
301  ef.pushElement("topicPermissions");
302  // For each of the topics in the request, add an entry
303  // to the response
304  Element topicsElement = msg.getElement(TOPICS);
305  for (size_t i = 0; i < topicsElement.numValues(); ++i) {
306  ef.appendElement();
307  ef.setElement("topic", topicsElement.getValueAsString(i));
308  ef.setElement("result", permission); //PERMISSIONED
309  ef.popElement();
310  }
311  ef.popElement();
312  session->sendResponse(response);
313  }
314  }
315  }
316  else {
317  MessageIterator iter(event);
318  while (iter.next()) {
319  Message msg = iter.message();
320  MutexGuard guard(&g_lock);
321  if (g_authorizationStatus.find(msg.correlationId()) !=
322  g_authorizationStatus.end()) {
323  if (msg.messageType() == AUTHORIZATION_SUCCESS) {
324  g_authorizationStatus[msg.correlationId()] = AUTHORIZED;
325  }
326  else {
327  g_authorizationStatus[msg.correlationId()] = FAILED;
328  }
329  }
330  msg.print(std::cout);
331  }
332  }
333 
334  return true;

423  bool authorize(const Service &authService,
424  Identity *providerIdentity,
425  ProviderSession *session,
426  const CorrelationId &cid)
427  {
428  {
429  MutexGuard guard(&g_lock);
430  g_authorizationStatus[cid] = WAITING;
431  }
432  EventQueue tokenEventQueue;
433  session->generateToken(CorrelationId(), &tokenEventQueue);
434  std::string token;
435  Event event = tokenEventQueue.nextEvent();
436  if (event.eventType() == Event::TOKEN_STATUS ||
437  event.eventType() == Event::REQUEST_STATUS) {
438  MessageIterator iter(event);
439  while (iter.next()) {
440  Message msg = iter.message();
441  {
442  MutexGuard guard(&g_lock);
443  msg.print(std::cout);
444  }
445  if (msg.messageType() == TOKEN_SUCCESS) {
446  token = msg.getElementAsString(TOKEN);
447  }
448  else if (msg.messageType() == TOKEN_FAILURE) {
449  break;
450  }
451  }
452  }
453  if (token.length() == 0) {
454  MutexGuard guard(&g_lock);
455  std::cout << "Failed to get token" << std::endl;
456  return false;
457  }
458 
459  Request authRequest = authService.createAuthorizationRequest();
460  authRequest.set(TOKEN, token.c_str());
461 
462  session->sendAuthorizationRequest(
463  authRequest,
464  providerIdentity,
465  cid);
466 
467  time_t startTime = time(0);
468  const int WAIT_TIME_SECONDS = 10;
469  while (true) {
470  {
471  MutexGuard guard(&g_lock);
472  if (WAITING != g_authorizationStatus[cid]) {
473  return AUTHORIZED == g_authorizationStatus[cid];
474  }
475  }
476  time_t endTime = time(0);
477  if (endTime - startTime > WAIT_TIME_SECONDS) {
478  return false;
479  }
480  SLEEP(1);
481  }
482  }
