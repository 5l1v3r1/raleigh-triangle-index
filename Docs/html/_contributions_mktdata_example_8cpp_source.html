223  }

226  MutexGuard guard(&g_lock);
227  std::cout << "Failed to get token" << std::endl;
228  return false;

232  authRequest.set(TOKEN, token.c_str());
233 
234  session->sendAuthorizationRequest(
235  authRequest,
236  providerIdentity,
237  cid);

239  time_t startTime = time(0);
240  const int WAIT_TIME_SECONDS = 10;
241  while (true) {
242  {
243  MutexGuard guard(&g_lock);
244  if (WAITING != g_authorizationStatus[cid]) {
245  return AUTHORIZED == g_authorizationStatus[cid];
246  }
